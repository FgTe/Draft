  \begin{aligned}
    M_{n+1} &= M_n - ( 2 \times D_{n-1} ) \\

  \end{aligned} \\

    N_{n+1} &= M_n - ( 2 \times D_{n-1} ) + m_{n-1} \\

  \end{aligned}
\end{gather*}
